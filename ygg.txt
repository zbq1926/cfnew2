// 工具函数依赖的变量提前声明
const byteToHex = Array.from({ length: 256 }, (_, i) => (i + 256).toString(16).slice(1));

// 配置初始化
let userID = 'ee760db6-9bf6-4130-b193-f822b88b5757';
const proxyIPs = ['127.0.0.1:443'];
const cn_hostnames = ['example.com'];
let CDNIP = 'www.via.com.sg';
let IP1 = 'www.via.com', IP2 = 'cis.via.com', IP3 = 'africa.via.com', IP4 = 'www.via.com.sg';
let IP5 = 'www.viaeurope.at', IP6 = 'www.via.com.mt', IP7 = 'qa.viamiddleeast.com';
let IP8 = 'usa.via.com', IP9 = 'myanmar.via.com', IP10 = 'www.via.com.tw';
let IP11 = 'www.viaeurope.ch', IP12 = 'www.via.com.br', IP13 = 'www.viasoutheasteurope.com';
const ports = ['80', '8080', '8880', '2052', '2082', '2086', '2095', '443', '8443', '2053', '2083', '2087', '2096'];
const dohURL = 'https://cloudflare-dns.com/dns-query';
let proxyIP = proxyIPs[Math.floor(Math.random() * proxyIPs.length)];
let proxyPort = proxyIP?.match(/:(\d+)$/)?.[1] || '443';
const wsEd = 2506;
const timeout = 30000;

// 工具函数定义
const isValidIP = ip => {
  const ipv4Regex = /^(?:(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)\.){3}(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)$/;
  const ipv6Regex = /^([0-9a-fA-F]{1,4}:){7,7}[0-9a-fA-F]{1,4}$|^([0-9a-fA-F]{1,4}:){1,7}:$/;
  return ipv4Regex.test(ip) || ipv6Regex.test(ip.replace(/\[|\]/g, ''));
};

const isValidUUID = u => /^[0-9a-f]{8}-[0-9a-f]{4}-[4][0-9a-f]{3}-[89ab][0-9a-f]{3}-[0-9a-f]{12}$/i.test(u);

const safeCloseWS = ws => {
  try {
    if (ws.readyState === WebSocket.OPEN || ws.readyState === WebSocket.CLOSING) {
      ws.close(1000, 'Normal closure');
    }
  } catch (e) {}
};

const base64ToBuf = b64 => {
  try {
    if (!b64) return null;
    b64 = b64.replace(/-/g, '+').replace(/_/g, '/');
    const padLength = 4 - (b64.length % 4);
    if (padLength < 4) b64 += '='.repeat(padLength);
    
    const binaryStr = atob(b64);
    const uint8Arr = new Uint8Array(binaryStr.length);
    for (let i = 0; i < binaryStr.length; i++) {
      uint8Arr[i] = binaryStr.charCodeAt(i);
    }
    return uint8Arr.buffer;
  } catch (e) {
    console.error('Base64 decode error:', e);
    return null;
  }
};

const bytesToUUID = bytes => {
  return [
    byteToHex[bytes[0]] + byteToHex[bytes[1]] + byteToHex[bytes[2]] + byteToHex[bytes[3]],
    byteToHex[bytes[4]] + byteToHex[bytes[5]],
    byteToHex[bytes[6]] + byteToHex[bytes[7]],
    byteToHex[bytes[8]] + byteToHex[bytes[9]],
    byteToHex[bytes[10]] + byteToHex[bytes[11]] + byteToHex[bytes[12]] + byteToHex[bytes[13]] + byteToHex[bytes[14]] + byteToHex[bytes[15]]
  ].join('-').toLowerCase();
};

// 核心功能函数
const parseXlessHeader = async (buf, uuid) => {
  try {
    if (buf.byteLength < 24) return { err: 'Header too short' };
    
    const uuidBytes = new Uint8Array(buf.slice(1, 17));
    const userUuidList = uuid.includes(',') ? uuid.split(',').map(u => u.trim()) : [uuid.trim()];
    const parsedUuid = bytesToUUID(uuidBytes);
    const isUuidValid = userUuidList.includes(parsedUuid) || await checkUuidInApiResponse(parsedUuid);
    if (!isUuidValid) return { err: `Invalid UUID: ${parsedUuid}` };
    
    const optLen = new Uint8Array(buf.slice(17, 18))[0];
    const cmd = new Uint8Array(buf.slice(18 + optLen, 19 + optLen))[0];
    const isUDP = cmd === 2;
    if (![1, 2].includes(cmd)) return { err: `Invalid command: ${cmd}` };
    
    const port = new DataView(buf.slice(19 + optLen, 21 + optLen)).getUint16(0, false);
    
    let addrIdx = 21 + optLen;
    const addrType = new Uint8Array(buf.slice(addrIdx, addrIdx + 1))[0];
    let addr = '', dataIdx = 0;
    
    switch (addrType) {
      case 1: // IPv4
        addr = new Uint8Array(buf.slice(addrIdx + 1, addrIdx + 5)).join('.');
        dataIdx = addrIdx + 5;
        break;
      case 2: // Domain
        const domainLen = new Uint8Array(buf.slice(addrIdx + 1, addrIdx + 2))[0];
        addr = new TextDecoder('utf-8').decode(buf.slice(addrIdx + 2, addrIdx + 2 + domainLen));
        dataIdx = addrIdx + 2 + domainLen;
        break;
      case 3: // IPv6
        const ipv6Bytes = new Uint8Array(buf.slice(addrIdx + 1, addrIdx + 17));
        addr = Array.from({ length: 8 }, (_, i) => {
          const chunk = (ipv6Bytes[i * 2] << 8) | ipv6Bytes[i * 2 + 1];
          return chunk.toString(16);
        }).join(':');
        dataIdx = addrIdx + 17;
        break;
      default:
        return { err: `Invalid address type: ${addrType}` };
    }
    
    return {
      err: null,
      addr,
      port,
      dataIdx,
      isUDP,
      ver: new Uint8Array(buf.slice(0, 1))
    };
  } catch (e) {
    return { err: `Parse error: ${e.message}` };
  }
};

const makeWSReadable = (ws, earlyData, log) => {
  let isCanceled = false;
  
  return new ReadableStream({
    start(controller) {
      if (earlyData) controller.enqueue(earlyData);
      
      const onMessage = e => {
        if (!isCanceled) controller.enqueue(e.data);
      };
      
      const onClose = () => {
        if (!isCanceled) controller.close();
      };
      
      const onError = e => {
        if (!isCanceled) controller.error(e);
      };
      
      ws.addEventListener('message', onMessage);
      ws.addEventListener('close', onClose);
      ws.addEventListener('error', onError);
      
      return () => {
        isCanceled = true;
        ws.removeEventListener('message', onMessage);
        ws.removeEventListener('close', onClose);
        ws.removeEventListener('error', onError);
      };
    },
    cancel() {
      isCanceled = true;
      safeCloseWS(ws);
    }
  });
};

const remoteToWS = (socket, ws, xlessHeader, log) => {
  let isHeaderSent = false;
  
  socket.readable.pipeTo(new WritableStream({
    write(chunk) {
      if (ws.readyState !== WebSocket.OPEN) throw new Error('WebSocket closed');
      const data = isHeaderSent ? chunk : new Blob([xlessHeader, chunk]).arrayBuffer();
      ws.send(data);
      isHeaderSent = true;
    },
    close() {
      log('Remote closed');
      safeCloseWS(ws);
    },
    abort(e) {
      log(`Remote error: ${e.message}`);
      safeCloseWS(ws);
    }
  })).catch(e => {
    log(`Forward error: ${e.message}`);
    safeCloseWS(ws);
  });
};

const handleTCP = async (remoteSocket, addr, port, data, ws, ver, log) => {
  const connectAndWrite = async (targetAddr, targetPort) => {
    if (isValidIP(targetAddr)) {
      targetAddr = targetAddr.includes(':') ? `[${targetAddr}]` : `www.${targetAddr}.sslip.io`;
    }
    
    log(`Connecting to ${targetAddr}:${targetPort}`);
    const socket = connect({ hostname: targetAddr, port: targetPort });
    const writer = socket.writable.getWriter();
    await writer.write(data);
    writer.releaseLock();
    return socket;
  };
  
  const retryWithProxy = async () => {
    if (!proxyIP) return;
    try {
      log(`Retrying with proxy ${proxyIP}:${proxyPort}`);
      const socket = await connectAndWrite(proxyIP, proxyPort);
      remoteSocket.value = socket;
      remoteToWS(socket, ws, new Uint8Array([ver[0], 0]), log);
    } catch (e) {
      log(`Proxy retry failed: ${e.message}`);
      safeCloseWS(ws);
    }
  };
  
  try {
    const socket = await connectAndWrite(addr, port);
    remoteSocket.value = socket;
    remoteToWS(socket, ws, new Uint8Array([ver[0], 0]), log);
    
    let hasData = false;
    socket.readable.pipeTo(new WritableStream({
      write() { hasData = true; }
    })).catch(() => {});
    
    socket.closed.catch(e => {
      log(`TCP error: ${e.message}`);
      if (proxyIP && !hasData) retryWithProxy();
    });
  } catch (e) {
    log(`Connection failed: ${e.message}`);
    if (proxyIP) retryWithProxy();
    else safeCloseWS(ws);
  }
};

const handleDNS = async (ws, ver, data, log) => {
  const xlessHeader = new Uint8Array([ver[0], 0]);
  let headerSent = false;
  
  const transform = new TransformStream({
    transform(chunk, ctl) {
      for (let i = 0; i < chunk.byteLength;) {
        const len = new DataView(chunk.slice(i, i + 2)).getUint16(0);
        ctl.enqueue(new Uint8Array(chunk.slice(i + 2, i + 2 + len)));
        i += 2 + len;
      }
    }
  });

  transform.readable.pipeTo(new WritableStream({
    async write(chunk) {
      try {
        const res = await fetch(dohURL, {
          method: 'POST',
          headers: { 'Content-Type': 'application/dns-message' },
          body: chunk,
          cf: { cacheTtl: 300 }
        });
        
        const buf = await res.arrayBuffer();
        const lenBuf = new Uint8Array([(buf.byteLength >> 8) & 0xff, buf.byteLength & 0xff]);
        const responseData = headerSent 
          ? new Blob([lenBuf, buf]).arrayBuffer()
          : new Blob([xlessHeader, lenBuf, buf]).arrayBuffer();
          
        ws.send(responseData);
        headerSent = true;
      } catch (e) {
        log(`DNS error: ${e.message}`);
        safeCloseWS(ws);
      }
    }
  })).catch(e => log(`DNS stream error: ${e.message}`));

  try {
    const writer = transform.writable.getWriter();
    await writer.write(data);
    writer.releaseLock();
  } catch (e) {
    log(`DNS write error: ${e.message}`);
  }
};

const xlessOverWS = async (request, ctx) => {
  const wsPair = new WebSocketPair();
  const [client, ws] = Object.values(wsPair);
  ws.accept();

  let addr = '', port = 0;
  const log = msg => console.log(`[${addr}:${port}] ${msg}`);
  const earlyData = base64ToBuf(request.headers.get('sec-websocket-protocol') || '');
  const readable = makeWSReadable(ws, earlyData, log);
  const remoteSocket = { value: null };

  const timeoutId = setTimeout(() => {
    log('Connection timed out');
    safeCloseWS(ws);
  }, timeout);
  ws.addEventListener('close', () => clearTimeout(timeoutId));

  readable.pipeTo(new WritableStream({
    async write(chunk) {
      if (remoteSocket.value) {
        try {
          const writer = remoteSocket.value.writable.getWriter();
          await writer.write(chunk);
          writer.releaseLock();
        } catch (e) {
          log(`Write failed: ${e.message}`);
          safeCloseWS(ws);
        }
        return;
      }

      const parsed = await parseXlessHeader(chunk, userID);
      if (parsed.err) {
        log(`Header error: ${parsed.err}`);
        safeCloseWS(ws);
        return;
      }
      
      addr = parsed.addr;
      port = parsed.port;
      
      if (parsed.isUDP) {
        if (port !== 53) {
          log('UDP only allowed for DNS (port 53)');
          safeCloseWS(ws);
          return;
        }
        return handleDNS(ws, parsed.ver, chunk.slice(parsed.dataIdx), log);
      }

      handleTCP(remoteSocket, addr, port, chunk.slice(parsed.dataIdx), ws, parsed.ver, log);
    },
    abort(e) {
      log(`Stream aborted: ${e.message}`);
      safeCloseWS(ws);
    }
  })).catch(e => {
    log(`Pipe error: ${e.message}`);
    safeCloseWS(ws);
  });

  ctx.waitUntil(new Promise(resolve => ws.addEventListener('close', resolve)));
  return new Response(null, { status: 101, webSocket: client });
};

// 辅助函数
async function checkUuidInApiResponse(targetUuid) {
  try {
    const apiResponse = await getApiResponse();
    return apiResponse?.users?.some(user => user.uuid === targetUuid) || false;
  } catch (error) {
    console.error('UUID validation error:', error);
    return false;
  }
}

async function getApiResponse() {
  return { users: [] };
}

const getXlessConfig = (uuid, hostName) => {
  const isWorkersDev = hostName.includes('workers.dev');
  const baseParams = `encryption=none&fp=randomized&type=ws&host=${hostName}&path=%2F%3Fed%3D${wsEd}`;
  const wsUrl = `xless://${uuid}@${CDNIP}:8880?${baseParams}&security=none#${hostName}`;
  const tlsUrl = `xless://${uuid}@${CDNIP}:8443?${baseParams}&security=tls&sni=${hostName}#${hostName}`;
  
  const ipList = [IP1, IP2, IP3, IP4, IP5, IP6, IP7, IP8, IP9, IP10, IP11, IP12, IP13];
  const allNodes = ipList.map((ip, i) => {
    const isTls = i >= 7;
    return `xless://${uuid}@${ip}:${ports[i]}?${baseParams}&security=${isTls ? 'tls&sni='+hostName : 'none'}#CF_V${i+1}_${ip}_${ports[i]}`;
  });
  const tlsNodes = allNodes.slice(7);
  
  const shareAll = btoa(allNodes.join('\n'));
  const shareTls = btoa(tlsNodes.join('\n'));
  
  return `
  <!DOCTYPE html>
  <html>
  <head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <link href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.3/dist/css/bootstrap.min.css" rel="stylesheet">
    <style>.limited-width { max-width: 500px; overflow: auto; word-wrap: break-word; }</style>
  </head>
  <body>
    <div class="container mt-4">
      <h1>Cloudflare Xless Proxy</h1>
      <hr>
      <p>ProxyIP: ${proxyIP}:${proxyPort}</p>
      <hr>
      
      ${isWorkersDev ? `
      <h3>1. WS节点（无加密）</h3>
      <div class="mb-3">
        <div class="limited-width mb-2">${wsUrl}</div>
        <button class="btn btn-primary" onclick="copy('${wsUrl}')">复制链接</button>
      </div>
      ` : ''}
      
      <h3>${isWorkersDev ? '2. ' : '1. '}WS+TLS节点（加密）</h3>
      <div class="mb-3">
        <div class="limited-width mb-2">${tlsUrl}</div>
        <button class="btn btn-primary" onclick="copy('${tlsUrl}')">复制链接</button>
      </div>
      
      <h3>${isWorkersDev ? '3. ' : '2. '}订阅链接</h3>
      <table class="table table-striped mb-3">
        <tbody>
          <tr>
            <td>通用分享链接</td>
            <td><button class="btn btn-primary" onclick="copy('${isWorkersDev ? shareAll : shareTls}')">复制</button></td>
          </tr>
          <tr>
            <td>通用订阅</td>
            <td class="limited-width">https://${hostName}/${uuid}/${isWorkersDev ? 'ty' : 'pty'}</td>
            <td><button class="btn btn-primary" onclick="copy('https://${hostName}/${uuid}/${isWorkersDev ? 'ty' : 'pty'}')">复制</button></td>
          </tr>
        </tbody>
      </table>
    </div>
    <script>
      function copy(text) {
        navigator.clipboard.writeText(text).then(() => alert('复制成功')).catch(() => {
          const el = document.createElement('textarea');
          el.value = text;
          document.body.appendChild(el);
          el.select();
          document.execCommand('copy');
          document.body.removeChild(el);
          alert('复制成功');
        });
      }
    </script>
  </body>
  </html>
  `;
};

const generateSubConfig = (uuid, hostName, isTlsOnly = false) => {
  const ipList = [IP1, IP2, IP3, IP4, IP5, IP6, IP7, IP8, IP9, IP10, IP11, IP12, IP13];
  const startIdx = isTlsOnly ? 7 : 0;
  const filteredIPs = ipList.slice(startIdx);
  const filteredPorts = ports.slice(startIdx);
  
  if (isTlsOnly ? uuid.includes('pty') : uuid.includes('ty')) {
    const nodes = filteredIPs.map((ip, i) => {
      const isTls = i + startIdx >= 7;
      const params = `encryption=none&fp=randomized&type=ws&host=${hostName}&path=%2F%3Fed%3D${wsEd}&security=${isTls ? 'tls&sni='+hostName : 'none'}`;
      return `xless://${uuid}@${ip}:${filteredPorts[i]}?${params}#CF_V${i+startIdx+1}_${ip}_${filteredPorts[i]}`;
    });
    return btoa(nodes.join('\n'));
  }
  
  return '';
};

// 主入口
export default {
  async fetch(request, env, ctx) {
    try {
      // 从环境变量加载配置
      userID = env.UUID || userID;
      proxyIP = env.PROXYIP || proxyIP;
      CDNIP = env.CDNIP || CDNIP;
      const envIPs = env.IP_LIST?.split(',') || [];
      if (envIPs.length === 13) {
        [IP1, IP2, IP3, IP4, IP5, IP6, IP7, IP8, IP9, IP10, IP11, IP12, IP13] = envIPs;
      }

      const url = new URL(request.url);
      const pathSegments = url.pathname.split('/').filter(Boolean);
      const upgradeHeader = request.headers.get('Upgrade');
      
      // 处理WebSocket连接
      if (upgradeHeader === 'websocket') {
        return await xlessOverWS(request, ctx);
      }
      
      // 处理配置页面和订阅
      const uuid = pathSegments[0] || '';
      const action = pathSegments[1] || '';
      const hostName = url.hostname;
      
      if (uuid && isValidUUID(uuid)) {
        if (['', 'ty', 'pty'].includes(action)) {
          const content = action === '' 
            ? getXlessConfig(uuid, hostName)
            : generateSubConfig(uuid, hostName, action.startsWith('p'));
          return new Response(content, { 
            headers: { 'Content-Type': action === '' ? 'text/html;charset=UTF-8' : 'text/plain;charset=UTF-8' } 
          });
        }
      }
      
      // 处理默认代理
      if (cn_hostnames.length > 0 && cn_hostnames[0] !== 'example.com') {
        const randomHost = cn_hostnames[Math.floor(Math.random() * cn_hostnames.length)];
        const newUrl = `https://${randomHost}${url.pathname}${url.search}`;
        const headers = new Headers(request.headers);
        headers.set('cf-connecting-ip', '1.2.3.4');
        headers.set('x-forwarded-for', '1.2.3.4');
        
        try {
          const response = await fetch(new Request(newUrl, {
            method: request.method,
            headers,
            body: request.body,
            redirect: 'manual'
          }));
          
          if ([301, 302, 307, 308].includes(response.status)) {
            return new Response('Redirect not allowed', { status: 403 });
          }
          return response;
        } catch (e) {
          return new Response(`Proxy error: ${e.message}`, { status: 502 });
        }
      }
      
      // 默认返回信息
      return new Response(JSON.stringify(request.cf, null, 2), {
        headers: { 'Content-Type': 'application/json' }
      });
    } catch (e) {
      return new Response(`Server error: ${e.message}`, { status: 500 });
    }
  }
};
