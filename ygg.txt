// 工具函数：IP验证（复用逻辑）
const isValidIP = ip => {
  const ipv4Regex = /^(?:(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)\.){3}(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)$/;
  const ipv6Regex = /^([0-9a-fA-F]{1,4}:){7,7}[0-9a-fA-F]{1,4}$/;
  return ipv4Regex.test(ip) || ipv6Regex.test(ip.replace(/\[|\]/g, ''));
};

// 工具函数：UUID验证
const isValidUUID = u => /^[0-9a-f]{8}-[0-9a-f]{4}-[4][0-9a-f]{3}-[89ab][0-9a-f]{3}-[0-9a-f]{12}$/i.test(u);

// 工具函数：安全关闭WebSocket
const safeCloseWS = ws => {
  try { if (ws.readyState === 1 || ws.readyState === 2) ws.close(1000, 'Normal closure'); } catch (e) {}
};

// 工具函数：Base64转Buffer（兼容处理）
const base64ToBuf = b64 => {
  try {
    if (!b64) return null;
    b64 = b64.replace(/-/g, '+').replace(/_/g, '/');
    const pad = b64.length % 4;
    if (pad) b64 += '='.repeat(4 - pad); // 补全Base64填充符
    return Uint8Array.from(atob(b64), c => c.charCodeAt(0)).buffer;
  } catch { return null; }
};

// UUID编码工具
const byteToHex = Array.from({length: 256}, (_, i) => (i + 256).toString(16).slice(1));
const bytesToUUID = bytes => [
  byteToHex[bytes[0]] + byteToHex[bytes[1]] + byteToHex[bytes[2]] + byteToHex[bytes[3]],

].join('-').toLowerCase();

// 解析Xless协议头
const parseXlessHeader = async (buf, uuid) => {
  try {
    if (buf.byteLength < 24) return { err: 'Header too short' };
    const slicedBuffer = new Uint8Array(buf.slice(1, 17));
    const uuids = uuid.includes(',') ? uuid.split(',') : [uuid];
    const isValidUser = uuids.some(u => bytesToUUID(slicedBuffer) === u.trim()) || await checkUuidInApiResponse(bytesToUUID(slicedBuffer));
    if (!isValidUser) return { err: 'Invalid UUID' };
    
    const optLen = new Uint8Array(buf.slice(17, 18))[0];
    const cmd = new Uint8Array(buf.slice(18 + optLen, 19 + optLen))[0];
    const isUDP = cmd === 2;
    if (![1, 2].includes(cmd)) return { err: `Invalid command: ${cmd}` };
    
    const port = new DataView(buf.slice(19 + optLen, 21 + optLen)).getUint16(0);
    let addrIdx = 21 + optLen;
    const addrType = new Uint8Array(buf.slice(addrIdx, addrIdx + 1))[0];
    let addr = '', dataIdx = 0;
    
    switch (addrType) {
      case 1: // IPv4
        addr = new Uint8Array(buf.slice(addrIdx + 1, addrIdx + 5)).join('.');
        dataIdx = addrIdx + 5;
        break;
      case 2: // 域名
        const len = new Uint8Array(buf.slice(addrIdx + 1, addrIdx + 2))[0];
        addr = new TextDecoder().decode(buf.slice(addrIdx + 2, addrIdx + 2 + len));
        dataIdx = addrIdx + 2 + len;
        break;
      case 3: // IPv6
        const dv = new DataView(buf.slice(addrIdx + 1, addrIdx + 17));
        addr = Array.from({length: 8}, (_, i) => dv.getUint16(i * 2).toString(16)).join(':');
        dataIdx = addrIdx + 17;
        break;
      default: return { err: `Invalid address type: ${addrType}` };
    }
    
    return {
      err: null,
      addr,
      port,
      dataIdx,
      isUDP,
      ver: new Uint8Array(buf.slice(0, 1))
    };
  } catch (e) {
    return { err: `Parse error: ${e.message}` };
  }
};

// 创建WebSocket可读流
const makeWSReadable = (ws, earlyData, log) => {
  let canceled = false;
  return new ReadableStream({
    start(ctl) {
      const onMsg = e => !canceled && ctl.enqueue(e.data);
      const onClose = () => {
        safeCloseWS(ws);
        !canceled && ctl.close();
      };
      ws.addEventListener('message', onMsg);
      ws.addEventListener('close', onClose);
      ws.addEventListener('error', e => ctl.error(e));
      if (earlyData) ctl.enqueue(earlyData);
      
      return () => {
        ws.removeEventListener('message', onMsg);
        ws.removeEventListener('close', onClose);
      };
    },
    cancel() {
      canceled = true;
      safeCloseWS(ws);
    }
  });
};

// 远程数据转发到WebSocket
const remoteToWS = (socket, ws, xlessHeader, log) => {
  let headerSent = false;
  socket.readable.pipeTo(new WritableStream({
    write(chunk) {
      if (ws.readyState !== 1) throw new Error('WebSocket closed');
      ws.send(headerSent ? chunk : new Blob([xlessHeader, chunk]).arrayBuffer());
      headerSent = true;
    },
    close() { log('Remote connection closed'); },
    abort(e) { log(`Remote error: ${e.message}`); safeCloseWS(ws); }
  })).catch(e => { log(`Forward error: ${e.message}`); safeCloseWS(ws); });
};

// 处理TCP连接
const handleTCP = async (remoteSocket, addr, port, data, ws, ver, log, ctx) => {
  async function connectAndWrite(address, port) {
    // 统一处理IP格式（IPv4/IPv6）
    if (isValidIP(address)) {
      address = address.includes(':') ? `[${address}]` : `www.${address}.sslip.io`;
    }
    log(`Connecting to ${address}:${port}`);
    const socket = connect({ hostname: address, port });
    const writer = socket.writable.getWriter();
    await writer.write(data);
    writer.releaseLock();
    return socket;
  }

  let hasIncomingData = false;
  const retry = async () => {
    try {
      const socket = await connectAndWrite(proxyIP || addr, proxyPort || port);
      socket.closed.catch(e => log(`Retry error: ${e.message}`)).finally(() => safeCloseWS(ws));
      remoteToWS(socket, ws, new Uint8Array([ver[0], 0]), log);
    } catch (e) {
      log(`Retry failed: ${e.message}`);
      safeCloseWS(ws);
    }
  };

  try {
    const socket = await connectAndWrite(addr, port);
    remoteSocket.value = socket;
    remoteToWS(socket, ws, new Uint8Array([ver[0], 0]), log);
    
    // 监听是否有数据流入
    socket.readable.pipeTo(new WritableStream({
      write() { hasIncomingData = true; }
    })).catch(() => {});
    
    socket.closed.catch(e => {
      log(`TCP error: ${e.message}`);
      if (proxyIP && !hasIncomingData) retry();
    });
  } catch (e) {
    log(`Connection failed: ${e.message}`);
    if (proxyIP) retry();
    else safeCloseWS(ws);
  }
};

// 处理DNS请求
const handleDNS = async (ws, ver, data, log) => {
  const xlessHeader = new Uint8Array([ver[0], 0]);
  let headerSent = false;
  
  const transform = new TransformStream({
    transform(chunk, ctl) {
      for (let i = 0; i < chunk.byteLength;) {
        const len = new DataView(chunk.slice(i, i + 2)).getUint16(0);
        ctl.enqueue(new Uint8Array(chunk.slice(i + 2, i + 2 + len)));
        i += 2 + len;
      }
    }
  });

  transform.readable.pipeTo(new WritableStream({
    async write(chunk) {
      try {
        const res = await fetch(dohURL, {
          method: 'POST',
          headers: { 'Content-Type': 'application/dns-message' },
          body: chunk,
          cf: { cacheTtl: 300 } // 缓存DNS结果5分钟
        });
        const buf = await res.arrayBuffer();
        const lenBuf = new Uint8Array([(buf.byteLength >> 8) & 0xff, buf.byteLength & 0xff]);
        ws.send(headerSent 
          ? new Blob([lenBuf, buf]).arrayBuffer()
          : new Blob([xlessHeader, lenBuf, buf]).arrayBuffer()
        );
        headerSent = true;
      } catch (e) {
        log(`DNS query failed: ${e.message}`);
        safeCloseWS(ws);
      }
    }
  })).catch(e => log(`DNS stream error: ${e.message}`));

  // 写入初始数据
  try {
    const writer = transform.writable.getWriter();
    await writer.write(data);
    writer.releaseLock();
  } catch (e) {
    log(`DNS write error: ${e.message}`);
  }
};

// WebSocket代理主逻辑
const xlessOverWS = async (request, ctx) => {
  const wsPair = new WebSocketPair();
  const [client, ws] = Object.values(wsPair);
  ws.accept();

  let addr = '', port = 0;
  const log = msg => console.log(`[${addr}:${port}] ${msg}`);
  const earlyData = base64ToBuf(request.headers.get('sec-websocket-protocol') || '');
  const readable = makeWSReadable(ws, earlyData, log);
  const remoteSocket = { value: null };

  // 超时控制
  const timeoutId = setTimeout(() => { 
    log('Connection timed out'); 
    safeCloseWS(ws); 
  }, timeout);
  ws.addEventListener('close', () => clearTimeout(timeoutId));

  readable.pipeTo(new WritableStream({
    async write(chunk) {
      if (remoteSocket.value) {
        try {
          const w = remoteSocket.value.writable.getWriter();
          await w.write(chunk);
          w.releaseLock();
        } catch (e) {
          log(`Write failed: ${e.message}`);
          safeCloseWS(ws);
        }
        return;
      }

      const parsed = await parseXlessHeader(chunk, userID);
      if (parsed.err) {
        log(`Header error: ${parsed.err}`);
        safeCloseWS(ws);
        return;
      }
      
      addr = parsed.addr;
      port = parsed.port;
      
      if (parsed.isUDP) {
        if (port !== 53) {
          log('UDP only allowed for DNS (port 53)');
          safeCloseWS(ws);
          return;
        }
        return handleDNS(ws, parsed.ver, chunk.slice(parsed.dataIdx), log);
      }

      handleTCP(remoteSocket, addr, port, chunk.slice(parsed.dataIdx), ws, parsed.ver, log, ctx);
    },
    abort(e) { log(`Stream aborted: ${e.message}`); safeCloseWS(ws); }
  })).catch(e => { log(`Pipe error: ${e.message}`); safeCloseWS(ws); });

  ctx.waitUntil(new Promise(resolve => ws.addEventListener('close', resolve)));
  return new Response(null, { status: 101, webSocket: client });
};

// 主入口
export default {
  async fetch(request, env, ctx) {
    try {
      // 从环境变量加载配置（优先级更高）
      userID = env.UUID || userID;
      proxyIP = env.PROXYIP || proxyIP;
      CDNIP = env.CDNIP || CDNIP;
      [IP1, IP2, IP3, IP4, IP5, IP6, IP7, IP8, IP9, IP10, IP11, IP12, IP13] = 
        env.IP_LIST?.split(',') || [IP1, IP2, IP3, IP4, IP5, IP6, IP7, IP8, IP9, IP10, IP11, IP12, IP13];

      const url = new URL(request.url);
      const pathSegments = url.pathname.split('/').filter(Boolean);
      const upgradeHeader = request.headers.get('Upgrade');
      
      // 处理WebSocket连接
      if (upgradeHeader === 'websocket') {
        return await xlessOverWS(request, ctx);
      }
      
      // 处理配置页面和订阅
      const uuid = pathSegments[0] || '';
      const action = pathSegments[1] || '';
      const hostName = url.hostname;
      
      if (uuid && isValidUUID(uuid)) {
        if (['', 'ty', 'cl', 'sb', 'pty', 'pcl', 'psb'].includes(action)) {
          const content = action === '' 
            ? getXlessConfig(uuid, hostName)
            : generateSubConfig(uuid, hostName, action.startsWith('p'));
          return new Response(content, { 
            headers: { 'Content-Type': action === '' ? 'text/html;charset=UTF-8' : 'text/plain;charset=UTF-8' } 
          });
        }
      }
      
      // 处理默认代理（仅当配置了有效主机名）
      if (cn_hostnames.length > 0 && cn_hostnames[0] !== 'example.com') {
        const randomHost = cn_hostnames[Math.floor(Math.random() * cn_hostnames.length)];
        const newUrl = `https://${randomHost}${url.pathname}${url.search}`;
        const headers = new Headers(request.headers);
        headers.set('cf-connecting-ip', '1.2.3.4');
        headers.set('x-forwarded-for', '1.2.3.4');
        headers.set('referer', 'https://www.google.com/search?q=hello');
        
        try {
          const response = await fetch(new Request(newUrl, {
            method: request.method,
            headers,
            body: request.body,
            redirect: 'manual' // 禁止自动重定向
          }));
          
          // 拦截重定向避免循环
          if ([301, 302, 307, 308].includes(response.status)) {
            return new Response('Redirect not allowed', { status: 403 });
          }
          return response;
        } catch (e) {
          return new Response(`Proxy error: ${e.message}`, { status: 502 });
        }
      }
      
      // 默认返回CF信息
      return new Response(JSON.stringify(request.cf, null, 2), {
        headers: { 'Content-Type': 'application/json' }
      });
    } catch (e) {
      return new Response(`Server error: ${e.message}`, { status: 500 });
    }
  }
};
