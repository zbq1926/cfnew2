// @ts-ignore
import { connect } from 'cloudflare:sockets';

const config = {
    uuid: 'd342d11e-d424-4583-b36e-524ab1f0afa4',
    proxyIP: '',
    wsEd: 2506,
    timeout: 30000
};

const isValidUUID = u => /^[0-9a-f]{8}-[0-9a-f]{4}-[4][0-9a-f]{3}-[89ab][0-9a-f]{3}-[0-9a-f]{12}$/i.test(u);
if (!isValidUUID(config.uuid)) throw new Error('Invalid UUID');

const safeCloseWS = ws => {
    try { if (ws.readyState === 1 || ws.readyState === 2) ws.close() } catch (e) {}
};

const base64ToBuf = b64 => {
    try {
        b64 = b64.replace(/-/g, '+').replace(/_/g, '/');
        return Uint8Array.from(atob(b64), c => c.charCodeAt(0)).buffer;
    } catch { return null }
};

const byteToHex = Array.from({length:256}, (_,i) => (i+256).toString(16).slice(1));
const bytesToUUID = bytes => [
    byteToHex[bytes[0]]+byteToHex[bytes[1]]+byteToHex[bytes[2]]+byteToHex[bytes[3]],
    byteToHex[bytes[4]]+byteToHex[bytes[5]],
    byteToHex[bytes[6]]+byteToHex[bytes[7]],
    byteToHex[bytes[8]]+byteToHex[bytes[9]],
    byteToHex[bytes[10]]+byteToHex[bytes[11]]+byteToHex[bytes[12]]+byteToHex[bytes[13]]+byteToHex[bytes[14]]+byteToHex[bytes[15]]
].join('-').toLowerCase();

const parseVlessHeader = (buf, uuid) => {
    try {
        if (buf.byteLength < 24) return { err: 'Too short' };
        
        if (bytesToUUID(new Uint8Array(buf.slice(1,17))) !== uuid) return { err: 'Bad UUID' };
        
        const optLen = new Uint8Array(buf.slice(17,18))[0];
        const cmd = new Uint8Array(buf.slice(18+optLen,19+optLen))[0];
        const isUDP = cmd === 2;
        if (![1,2].includes(cmd)) return { err: `Bad cmd: ${cmd}` };
        
        const port = new DataView(buf.slice(19+optLen,21+optLen)).getUint16(0);
        
        let addrIdx = 21 + optLen;
        const addrType = new Uint8Array(buf.slice(addrIdx, addrIdx+1))[0];
        let addr = '', dataIdx = 0;
        
        switch(addrType) {
            case 1:
                addr = new Uint8Array(buf.slice(addrIdx+1, addrIdx+5)).join('.');
                dataIdx = addrIdx + 5;
                break;
            case 2:
                const len = new Uint8Array(buf.slice(addrIdx+1, addrIdx+2))[0];
                addr = new TextDecoder().decode(buf.slice(addrIdx+2, addrIdx+2+len));
                dataIdx = addrIdx + 2 + len;
                break;
            case 3:
                const dv = new DataView(buf.slice(addrIdx+1, addrIdx+17));
                addr = Array.from({length:8}, (_,i) => dv.getUint16(i*2).toString(16)).join(':');
                dataIdx = addrIdx + 17;
                break;
            default: return { err: `Bad addr type: ${addrType}` };
        }
        
        return {
            err: null,
            addr,
            port,
            dataIdx,
            isUDP,
            ver: new Uint8Array(buf.slice(0,1))
        };
    } catch (e) {
        return { err: e.message };
    }
};

const makeWSReadable = (ws, earlyData, log) => {
    let canceled = false;
    return new ReadableStream({
        start(ctl) {
            const onMsg = e => !canceled && ctl.enqueue(e.data);
            const onClose = () => {
                safeCloseWS(ws);
                !canceled && ctl.close();
            };
            ws.addEventListener('message', onMsg);
            ws.addEventListener('close', onClose);
            ws.addEventListener('error', e => ctl.error(e));
            if (earlyData) ctl.enqueue(earlyData);
            
            return () => {
                ws.removeEventListener('message', onMsg);
                ws.removeEventListener('close', onClose);
            };
        },
        cancel() {
            canceled = true;
            safeCloseWS(ws);
        }
    });
};

const remoteToWS = (socket, ws, vlessHeader, log) => {
    let headerSent = false;
    socket.readable.pipeTo(new WritableStream({
        write(chunk) {
            if (ws.readyState !== 1) throw new Error('WS closed');
            ws.send(headerSent ? chunk : new Blob([vlessHeader, chunk]).arrayBuffer());
            headerSent = true;
        },
        close() { log('Remote closed'); },
        abort(e) { log(`Remote error: ${e}`); safeCloseWS(ws); }
    })).catch(e => { log(`Forward error: ${e}`); safeCloseWS(ws); });
};

const handleTCP = (remoteSocket, addr, port, data, ws, ver, log, ctx) => {
    const connect = async (targetAddr, targetPort) => {
        log(`Connecting to ${targetAddr}:${targetPort}`);
        const socket = connect({ hostname: targetAddr, port: targetPort });
        const writer = socket.writable.getWriter();
        await writer.write(data);
        writer.releaseLock();
        return socket;
    };

    connect(addr, port).then(socket => {
        remoteSocket.value = socket;
        remoteToWS(socket, ws, new Uint8Array([ver[0], 0]), log);
        socket.closed.catch(e => {
            log(`TCP error: ${e}`);
            if (config.proxyIP) connect(config.proxyIP, port).then(s => {
                remoteSocket.value = s;
                remoteToWS(s, ws, new Uint8Array([ver[0], 0]), log);
            });
        });
    }).catch(e => {
        log(`Connect failed: ${e}`);
        if (config.proxyIP) connect(config.proxyIP, port).then(s => {
            remoteSocket.value = s;
            remoteToWS(s, ws, new Uint8Array([ver[0], 0]), log);
        }).catch(e => { log(`Retry failed: ${e}`); safeCloseWS(ws); });
    });
};

const handleDNS = async (ws, ver, data, log) => {
    const vlessHeader = new Uint8Array([ver[0], 0]);
    let headerSent = false;
    
    const transform = new TransformStream({
        transform(chunk, ctl) {
            for (let i = 0; i < chunk.byteLength;) {
                const len = new DataView(chunk.slice(i, i+2)).getUint16(0);
                ctl.enqueue(new Uint8Array(chunk.slice(i+2, i+2+len)));
                i += 2 + len;
            }
        }
    });

    transform.readable.pipeTo(new WritableStream({
        async write(chunk) {
            const res = await fetch('https://1.1.1.1/dns-query', {
                method: 'POST',
                headers: { 'Content-Type': 'application/dns-message' },
                body: chunk
            });
            const buf = await res.arrayBuffer();
            const lenBuf = new Uint8Array([(buf.byteLength>>8)&0xff, buf.byteLength&0xff]);
            ws.send(headerSent 
                ? new Blob([lenBuf, buf]).arrayBuffer()
                : new Blob([vlessHeader, lenBuf, buf]).arrayBuffer()
            );
            headerSent = true;
        }
    }));

    transform.writable.getWriter().write(data);
};

const vlessOverWS = (request, ctx) => {
    const wsPair = new WebSocketPair();
    const [client, ws] = Object.values(wsPair);
    ws.accept();

    let addr = '', port = 0;
    const log = msg => console.log(`[${addr}:${port}] ${msg}`);
    const earlyData = base64ToBuf(request.headers.get('sec-websocket-protocol') || '');
    const readable = makeWSReadable(ws, earlyData, log);
    const remoteSocket = { value: null };

    const timeout = setTimeout(() => { log('Timeout'); safeCloseWS(ws); }, config.timeout);
    ws.addEventListener('close', () => clearTimeout(timeout));

    readable.pipeTo(new WritableStream({
        async write(chunk) {
            if (remoteSocket.value) {
                const w = remoteSocket.value.writable.getWriter();
                await w.write(chunk);
                w.releaseLock();
                return;
            }

            const parsed = parseVlessHeader(chunk, config.uuid);
            if (parsed.err) throw new Error(parsed.err);
            
            addr = parsed.addr;
            port = parsed.port;
            
            if (parsed.isUDP) {
                if (port !== 53) throw new Error('UDP only for DNS');
                return handleDNS(ws, parsed.ver, chunk.slice(parsed.dataIdx), log);
            }

            handleTCP(remoteSocket, addr, port, chunk.slice(parsed.dataIdx), ws, parsed.ver, log, ctx);
        },
        abort(e) { log(`Abort: ${e}`); safeCloseWS(ws); }
    })).catch(e => { log(`Pipe error: ${e}`); safeCloseWS(ws); });

    ctx.waitUntil(new Promise(resolve => ws.addEventListener('close', resolve)));
    return new Response(null, { status: 101, webSocket: client });
};

const getConfig = (uuid, host) => {
    const path = `/?ed=${config.wsEd}`;
    const link = `vless://${uuid}@${host}:443?encryption=none&security=tls&sni=${host}&fp=randomized&type=ws&host=${host}&path=${encodeURIComponent(path)}#${host}`;
    return `V2Ray: ${link}\n\nClash:\n- type: vless\n  name: ${host}\n  server: ${host}\n  port: 443\n  uuid: ${uuid}\n  network: ws\n  tls: true\n  ws-opts:\n    path: "${path}"\n    headers: {host: ${host}}`;
};

export default {
    async fetch(request, env, ctx) {
        try {
            const uuid = env.UUID || config.uuid;
            const proxyIP = env.PROXYIP || config.proxyIP;
            const upgrade = request.headers.get('Upgrade');

            if (upgrade === 'websocket') return vlessOverWS(request, ctx);

            const url = new URL(request.url);
            if (url.pathname === '/') return new Response('OK', { status: 200 });
            if (url.pathname === `/${uuid}`) return new Response(getConfig(uuid, request.headers.get('Host')), {
                headers: { 'Content-Type': 'text/plain' }
            });

            return new Response('Not found', { status: 404 });
        } catch (e) {
            return new Response(e.message, { status: 500 });
        }
    }
};
