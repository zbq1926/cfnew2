// @ts-ignore
import { connect } from 'cloudflare:sockets';

// 配置参数
let config = {
    uuid: '2dd1340b-4b41-4aa9-95cb-e214754383ed',
    proxyIP: 'ProxyIP.US.CMLiussss.net',
    debug: false,
    timeout: 30000
};

// UUID验证
function isValidUUID(uuid) {
    const regex = /^[0-9a-f]{8}-[0-9a-f]{4}-[4][0-9a-f]{3}-[89ab][0-9a-f]{3}-[0-9a-f]{12}$/i;
    return regex.test(uuid);
}

// 初始化验证
if (!isValidUUID(config.uuid)) throw new Error('Invalid UUID');

// 日志工具
const logger = {
    log: (message, ...args) => config.debug && console.log(`[LOG] ${message}`, ...args),
    error: (message, ...args) => console.error(`[ERROR] ${message}`, ...args)
};

export default {
    async fetch(request, env, ctx) {
        try {
            // 从环境变量加载配置
            config.uuid = env.UUID || config.uuid;
            config.proxyIP = env.PROXY_IP || config.proxyIP;
            config.debug = env.DEBUG === 'true' || config.debug;

            const upgradeHeader = request.headers.get('Upgrade');
            
            // 健康检查路由
            const url = new URL(request.url);
            if (url.pathname === '/health') {
                return new Response(JSON.stringify({ status: 'ok', time: new Date().toISOString() }), {
                    headers: { 'Content-Type': 'application/json' }
                });
            }

            // WebSocket代理处理
            if (upgradeHeader === 'websocket') {
                return await handleWebSocket(request, ctx);
            }

            // 非WebSocket请求
            return new Response('Not found', { status: 404 });
        } catch (err) {
            logger.error('Main fetch error', err);
            return new Response(`Server error: ${err.message}`, { status: 500 });
        }
    }
};

// WebSocket处理
async function handleWebSocket(request, ctx) {
    // 创建WebSocket对
    const webSocketPair = new WebSocketPair();
    const [clientWs, serverWs] = Object.values(webSocketPair);
    serverWs.accept();

    // 连接元数据
    const connectionMeta = {
        remoteAddr: '',
        remotePort: 0,
        protocol: 'tcp',
        startTime: Date.now()
    };

    // 超时控制
    const timeoutId = setTimeout(() => {
        logger.log('Connection timed out', connectionMeta);
        safeCloseWebSocket(serverWs, 408, 'Timeout');
    }, config.timeout);

    // 清理函数
    const cleanup = () => {
        clearTimeout(timeoutId);
        const duration = Date.now() - connectionMeta.startTime;
        logger.log(`Connection closed. Duration: ${duration}ms`, connectionMeta);
    };

    // 监听WebSocket关闭
    serverWs.addEventListener('close', cleanup);
    serverWs.addEventListener('error', (err) => {
        logger.error('WebSocket error', err);
        cleanup();
    });

    // 处理早期数据
    const earlyData = parseEarlyData(request.headers.get('sec-websocket-protocol') || '');
    
    // 创建可读流
    const readableStream = createWebSocketReadableStream(serverWs, earlyData, logger);
    
    // 远程连接管理
    let remoteSocket = { value: null };

    // 客户端数据转发到远程
    const forwardPromise = readableStream.pipeTo(new WritableStream({
        async write(chunk) {
            if (remoteSocket.value) {
                return writeToSocket(remoteSocket.value, chunk);
            }

            // 解析VLESS头部
            const parseResult = parseVlessHeader(chunk, config.uuid);
            if (parseResult.error) {
                throw new Error(`Header parse error: ${parseResult.message}`);
            }

            // 更新连接元数据
            connectionMeta.remoteAddr = parseResult.address;
            connectionMeta.remotePort = parseResult.port;
            connectionMeta.protocol = parseResult.isUDP ? 'udp' : 'tcp';
            logger.log('Parsed connection metadata', connectionMeta);

            // 仅支持TCP
            if (parseResult.isUDP) {
                throw new Error('UDP is not supported');
            }

            // 建立TCP连接
            await establishTCPConnection(
                remoteSocket,
                parseResult.address,
                parseResult.port,
                chunk.slice(parseResult.dataIndex),
                serverWs,
                ctx
            );
        },
        close() {
            logger.log('Client stream closed');
            safeCloseSocket(remoteSocket.value);
        },
        abort(reason) {
            logger.error('Client stream aborted', reason);
            safeCloseSocket(remoteSocket.value);
            safeCloseWebSocket(serverWs, 1011, 'Stream error');
        }
    })).catch(err => {
        logger.error('Forwarding error', err);
        safeCloseWebSocket(serverWs, 1011, err.message);
    });

    // 绑定到上下文防止提前关闭
    ctx.waitUntil(forwardPromise);

    return new Response(null, {
        status: 101,
        webSocket: clientWs
    });
}

// 建立TCP连接
async function establishTCPConnection(remoteSocket, address, port, initialData, ws, ctx) {
    // 连接函数
    const connectToRemote = async (targetAddress, targetPort) => {
        try {
            logger.log(`Connecting to ${targetAddress}:${targetPort}`);
            const socket = connect({ hostname: targetAddress, port: targetPort });
            
            // 验证连接
            await Promise.race([
                socket.closed.then(() => { throw new Error('Connection closed immediately'); }),
                new Promise(resolve => setTimeout(resolve, 1000))
            ]);

            // 发送初始数据
            if (initialData.byteLength > 0) {
                await writeToSocket(socket, initialData);
            }
            
            return socket;
        } catch (err) {
            logger.error(`Connection to ${targetAddress}:${targetPort} failed`, err);
            throw err;
        }
    };

    // 重试逻辑
    const retryWithProxy = async () => {
        if (!config.proxyIP) {
            throw new Error('No proxy IP configured for retry');
        }
        return connectToRemote(config.proxyIP, port);
    };

    // 建立初始连接
    try {
        remoteSocket.value = await connectToRemote(address, port);
    } catch (err) {
        logger.log('Retrying with proxy IP');
        remoteSocket.value = await retryWithProxy();
    }

    // 远程数据转发到WebSocket
    const remoteToWsPromise = forwardSocketToWebSocket(remoteSocket.value, ws, logger);
    ctx.waitUntil(remoteToWsPromise);

    // 监听远程关闭
    remoteSocket.value.closed.catch(err => {
        logger.error('Remote socket closed with error', err);
        safeCloseWebSocket(ws, 1011, 'Remote closed');
    });
}

// 创建WebSocket可读流
function createWebSocketReadableStream(ws, earlyData, logger) {
    let isCanceled = false;

    return new ReadableStream({
        start(controller) {
            // 处理消息
            const handleMessage = (event) => {
                if (isCanceled) return;
                controller.enqueue(event.data);
            };

            // 处理关闭
            const handleClose = () => {
                if (isCanceled) return;
                logger.log('WebSocket closed by client');
                controller.close();
            };

            // 处理错误
            const handleError = (err) => {
                if (isCanceled) return;
                logger.error('WebSocket error', err);
                controller.error(err);
            };

            // 注册事件
            ws.addEventListener('message', handleMessage);
            ws.addEventListener('close', handleClose);
            ws.addEventListener('error', handleError);

            // 处理早期数据
            if (earlyData) {
                logger.log('Processing early data');
                controller.enqueue(earlyData);
            }

            // 清理函数
            return () => {
                ws.removeEventListener('message', handleMessage);
                ws.removeEventListener('close', handleClose);
                ws.removeEventListener('error', handleError);
            };
        },
        cancel(reason) {
            isCanceled = true;
            logger.log(`Readable stream canceled: ${reason}`);
        }
    });
}

// 解析VLESS头部
function parseVlessHeader(buffer, expectedUUID) {
    try {
        if (buffer.byteLength < 24) {
            return { error: true, message: 'Buffer too short' };
        }

        // 解析版本
        const version = new Uint8Array(buffer.slice(0, 1))[0];
        if (version !== 0) {
            return { error: true, message: `Unsupported version: ${version}` };
        }

        // 验证UUID
        const uuidBytes = new Uint8Array(buffer.slice(1, 17));
        const uuid = bytesToUUID(uuidBytes);
        if (uuid !== expectedUUID) {
            return { error: true, message: 'Invalid UUID' };
        }

        // 解析选项长度
        const optLength = new Uint8Array(buffer.slice(17, 18))[0];
        const optEnd = 18 + optLength;

        // 解析命令
        const command = new Uint8Array(buffer.slice(optEnd, optEnd + 1))[0];
        const isUDP = command === 2;
        if (![1, 2].includes(command)) {
            return { error: true, message: `Unsupported command: ${command}` };
        }

        // 解析端口
        const portStart = optEnd + 1;
        const port = new DataView(buffer.slice(portStart, portStart + 2)).getUint16(0);

        // 解析地址
        let addrStart = portStart + 2;
        const addrType = new Uint8Array(buffer.slice(addrStart, addrStart + 1))[0];
        let address = '';
        let addrLength = 0;
        let dataIndex = 0;

        switch (addrType) {
            case 1: // IPv4
                addrLength = 4;
                address = new Uint8Array(buffer.slice(addrStart + 1, addrStart + 1 + addrLength)).join('.');
                dataIndex = addrStart + 1 + addrLength;
                break;
            case 2: // 域名
                addrLength = new Uint8Array(buffer.slice(addrStart + 1, addrStart + 2))[0];
                address = new TextDecoder().decode(buffer.slice(addrStart + 2, addrStart + 2 + addrLength));
                dataIndex = addrStart + 2 + addrLength;
                break;
            case 3: // IPv6
                addrLength = 16;
                const ipv6Bytes = new DataView(buffer.slice(addrStart + 1, addrStart + 1 + addrLength));
                const ipv6Parts = [];
                for (let i = 0; i < 8; i++) {
                    ipv6Parts.push(ipv6Bytes.getUint16(i * 2).toString(16));
                }
                address = ipv6Parts.join(':');
                dataIndex = addrStart + 1 + addrLength;
                break;
            default:
                return { error: true, message: `Invalid address type: ${addrType}` };
        }

        return {
            error: false,
            address,
            port,
            dataIndex,
            isUDP,
            version
        };
    } catch (err) {
        return { error: true, message: err.message };
    }
}

// 转发Socket到WebSocket
async function forwardSocketToWebSocket(socket, ws, logger) {
    let hasData = false;
    try {
        await socket.readable.pipeTo(new WritableStream({
            write(chunk) {
                hasData = true;
                if (ws.readyState !== WebSocket.OPEN) {
                    throw new Error('WebSocket is not open');
                }
                ws.send(chunk);
            },
            close() {
                logger.log('Remote socket closed normally');
                safeCloseWebSocket(ws, 1000, 'Remote closed');
            },
            abort(reason) {
                logger.error('Remote socket stream aborted', reason);
                safeCloseWebSocket(ws, 1011, 'Remote error');
            }
        }));
    } catch (err) {
        logger.error('Socket to WebSocket forward error', err);
        safeCloseWebSocket(ws, 1011, err.message);
    }
    return hasData;
}

// 工具函数：写入Socket
async function writeToSocket(socket, data) {
    const writer = socket.writable.getWriter();
    try {
        await writer.write(data);
    } finally {
        writer.releaseLock();
    }
}

// 工具函数：安全关闭WebSocket
function safeCloseWebSocket(ws, code = 1000, reason = '') {
    try {
        if (ws.readyState === WebSocket.OPEN || ws.readyState === WebSocket.CLOSING) {
            ws.close(code, reason);
        }
    } catch (err) {
        logger.error('Failed to close WebSocket', err);
    }
}

// 工具函数：安全关闭Socket
function safeCloseSocket(socket) {
    if (socket) {
        try {
            socket.close();
        } catch (err) {
            logger.error('Failed to close socket', err);
        }
    }
}

// 工具函数：解析早期数据
function parseEarlyData(header) {
    if (!header) return null;
    try {
        // 处理URL安全的Base64
        const base64 = header.replace(/-/g, '+').replace(/_/g, '/');
        const decoded = atob(base64);
        return Uint8Array.from(decoded, c => c.charCodeAt(0)).buffer;
    } catch (err) {
        logger.error('Failed to parse early data', err);
        return null;
    }
}

// 工具函数：字节转UUID
const byteToHex = [];
for (let i = 0; i < 256; i++) {
    byteToHex.push((i + 256).toString(16).slice(1));
}

function bytesToUUID(bytes) {
    return `${byteToHex[bytes[0]]}${byteToHex[bytes[1]]}${byteToHex[bytes[2]]}${byteToHex[bytes[3]]}-${
           byteToHex[bytes[4]]}${byteToHex[bytes[5]]}-${
           byteToHex[bytes[6]]}${byteToHex[bytes[7]]}-${
           byteToHex[bytes[8]]}${byteToHex[bytes[9]]}-${
           byteToHex[bytes[10]]}${byteToHex[bytes[11]]}${byteToHex[bytes[12]]}${byteToHex[bytes[13]]}${byteToHex[bytes[14]]}${byteToHex[bytes[15]]}`.toLowerCase();
}
